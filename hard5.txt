void get_key(void)                
{
	unsigned int  i;	
	flag = 0x00;
    IO1PIN=0x000f0000;        
	while(1)
	{		
    	for(row=0X00;row<0X04;row++)	
        {					
        	if( row == 0X00)
            { 
            	temp3=0x00700000;   
		    }
            else if(row == 0X01)
            { 
            	temp3=0x00B00000;  
		    }
		    else if(row == 0X02)
		    {
            	temp3=0x00D00000;  
		    }
            else if(row == 0X03)
		    {
            	temp3=0x00E00000;   
		    }	
	  	  	var1 = temp3;
          	IO1PIN = var1;		
          	IO1CLR =~var1;		
          	scan();
          	delay(100);			
          	if(flag == 0xff) 
          	break;
       	} 
		if(flag == 0xff)
		break;
	}  
  
    for(i=0;i<16;i++)
    {
    	if(scan_code[i] == res1)     
    	{
			result =  ASCII_CODE[i];    
			break;                   
      	}
    }
}

void scan(void)
{	
    unsigned long int t;
    temp2 = IO1PIN;                             
    temp2 = temp2 & 0x000F0000;               
    if(temp2 != 0x000F0000)                     
    {
    	delay(1000);                              
    	temp2 = IO1PIN; 		
      	temp2 = temp2 & 0x000F0000;             

        if(temp2 != 0x000F0000)                
        {
        	flag = 0xff;
          	res1 = temp2; 
          	t = (temp3 & 0x00F00000);           
          	res1 = res1 | t;                 
        }
        else
        {
        	flag = 0x00;
        }
    }
}
